模块化简介

模块化简介

commonjs

commonjsLib.js

function sum(a,b) {
  return a+b;
}

function dem(a,b){
  return a-b;
}

module.exports = {
  sum,
  dem
}

main.js

const {sum,dem} = require('./commonjsLib.js');

const res = sum(10,20);
console.log(res);

console.log(dem(30, 10));

ES6

es6Lib.js

function sum(a,b) {
  return a+b;
}

function dem(a,b){
  return a-b;
}

export {
  sum,dem
}

main.js

const {sum,dem} = require('./commonjsLib.js');

import {sum,dem} from './es6Lib.js';

console.log(sum(10,20));
console.log(dem(30, 10));
